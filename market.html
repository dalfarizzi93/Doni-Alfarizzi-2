// DATA MASTER let products = [ { id: 1, name: "Midnight Rose", price: 150000, cat: "Wanita", stock: 12, img: "🌹" }, { id: 2, name: "Blue Ocean", price: 175000, cat: "Pria", stock: 8, img: "🌊" }, { id: 3, name: "Vanilla Sky", price: 125000, cat: "Wanita", stock: 20, img: "🍦" }, { id: 4, name: "Dark Coffee", price: 210000, cat: "Unisex", stock: 5, img: "☕" } ]; let cart = []; // FUNGSI LOGIN (Hanya dipicu saat klik Admin) function login() { const user = document.getElementById('username').value.trim(); if(!user) return alert("Masukkan username!"); if(user.toLowerCase() === 'admin') { document.getElementById('nav-admin').classList.remove('hidden'); document.getElementById('login-modal').classList.add('hidden'); showPage('admin'); alert("Akses Admin Diterima."); } else { alert("Username salah! Hanya admin yang bisa masuk."); } renderAll(); } // Tampilkan modal login hanya jika ingin ke halaman admin function showPage(page) { if(page === 'admin' && document.getElementById('nav-admin').classList.contains('hidden')) { document.getElementById('login-modal').classList.remove('hidden'); return; } document.getElementById('user-page').classList.add('hidden'); document.getElementById('admin-page').classList.add('hidden'); document.getElementById(page + '-page').classList.remove('hidden'); } function logout() { location.reload(); } // ADMIN EDIT FUNCTIONS function updateStock(id, amount) { const product = products.find(p => p.id === id); if (product.stock + amount < 0) return; product.stock += amount; renderAll(); } function updatePrice(id, newPrice) { const product = products.find(p => p.id === id); product.price = parseInt(newPrice) || 0; renderAll(); } function updateName(id, newName) { const product = products.find(p => p.id === id); product.name = newName; renderAll(); } // RENDER ENGINE function renderAll() { renderUserProducts(); renderAdminTable(); } function renderUserProducts() { const container = document.getElementById('product-list'); container.innerHTML = products.map(p => `
${p.img}

${p.cat}

${p.name}

Stok: ${p.stock}

Rp ${p.price.toLocaleString('id-ID')}

`).join(''); } function renderAdminTable() { const tableBody = document.getElementById('admin-table-body'); tableBody.innerHTML = products.map(p => `
${p.stock}
${p.stock > 5 ? 'Aman' : 'Restock'} `).join(''); } // SISTEM KERANJANG function addToCart(id) { const p = products.find(i => i.id === id); cart.push({...p}); updateCartUI(); } function updateCartUI() { document.getElementById('cart-count').innerText = cart.length; const container = document.getElementById('cart-items'); const totalText = document.getElementById('cart-total'); let total = 0; if (cart.length === 0) { container.innerHTML = '

Tas masih kosong.

'; totalText.innerText = "Rp 0"; return; } container.innerHTML = cart.map((item, i) => { total += item.price; return `
${item.img}

${item.name}

Rp ${item.price.toLocaleString('id-ID')}

`; }).join(''); totalText.innerText = "Rp " + total.toLocaleString('id-ID'); } function removeFromCart(i) { cart.splice(i, 1); updateCartUI(); } function toggleCart() { document.getElementById('cart-modal').classList.toggle('hidden'); } // CHECKOUT KE WHATSAPP function checkout() { if(cart.length === 0) return alert("Pilih produk dulu!"); const phone = "62881022362914"; // Format internasional tanpa simbol + let total = 0; // Membuat daftar barang untuk pesan let itemsList = cart.map((item, index) => { total += item.price; return `${index + 1}. *${item.name}* - Rp ${item.price.toLocaleString('id-ID')}`; }).join('%0A'); // %0A adalah kode untuk baris baru (enter) di URL const message = `Halo Admin PARFUM.ID,%0A%0ASaya ingin memesan produk berikut:%0A${itemsList}%0A%0A*Total Pembayaran: Rp ${total.toLocaleString('id-ID')}*%0A%0AMohon info detail pembayarannya ya, terima kasih!`; window.open(`https://wa.me/${phone}?text=${message}`, '_blank'); // Reset keranjang setelah checkout cart = []; updateCartUI(); toggleCart(); } // Pastikan modal login tertutup saat start window.onload = () => { document.getElementById('login-modal').classList.add('hidden'); renderAll(); };
